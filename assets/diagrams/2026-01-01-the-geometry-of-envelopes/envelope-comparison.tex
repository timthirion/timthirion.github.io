\documentclass[tikz,border=10pt]{standalone}
\usepackage{pgfplots}
\usepackage{etoolbox}
\usetikzlibrary{backgrounds}
\pgfplotsset{compat=1.18}

% Load color scheme
\ifdefstring{\theme}{latte}{
  \input{../../../_tikz/colors-latte}
}{
  \input{../../../_tikz/colors-mocha}
}

\begin{document}
\begin{tikzpicture}
  \begin{axis}[
      width=10cm,
      height=10cm,
      xmin=0, xmax=10,
      ymin=0, ymax=10,
      axis lines=center,
      axis line style={textColor, thick},
      tick style={textColor},
      tick label style={textColor},
      grid=major,
      grid style={gridColor, thin},
      xlabel={$x$},
      ylabel={$y$},
      xlabel style={textColor},
      ylabel style={textColor},
      legend pos=north east,
      legend style={draw=textColor, fill=bgColor, fill opacity=0.8, text opacity=1, font=\color{textColor}},
      legend cell align=left,
    ]

    % Draw the 10 line segments (lighter, in background) - IGNORE FOR LEGEND
    \foreach \a in {1,2,...,10} {
      \addplot[secondaryColor, thick, opacity=0.3, forget plot] coordinates {
        (0, \a) (10-\a, 0)
      };
    }

    % Envelope curve: y = (sqrt(10) - sqrt(x))^2
    \addplot[primaryColor, very thick, domain=0:10, samples=200, forget plot] {
      (sqrt(10) - sqrt(x))^2
    };

    % Parabola: y = 0.1x^2 - 2x + 10
    \addplot[accentColor, very thick, dashed, domain=0:10, samples=200, forget plot] {
      0.1*x^2 - 2*x + 10
    };

    % Quarter circle: center (10,10), radius 10
    \addplot[greenColor, very thick, dotted, domain=0:10, samples=200, forget plot] {
      10 - sqrt(100 - (x-10)^2)
    };

    % Now manually add legend entries
    \addlegendimage{primaryColor, very thick, solid}
    \addlegendentry{Envelope}

    \addlegendimage{accentColor, very thick, dashed}
    \addlegendentry{Parabola}

    \addlegendimage{greenColor, very thick, dotted}
    \addlegendentry{Circle}

  \end{axis}
\end{tikzpicture}
\end{document}
